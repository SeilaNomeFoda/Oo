function topos(p546)
    -- upvalues: (ref) vu529
    local vu547 = game.Players.LocalPlayer

    -- checagens iniciais: p546 e player/character/hrp vlidos
    if not p546 then
        return
    end
    if not vu547 or not vu547.Character then
        return
    end
    local hrp = vu547.Character:FindFirstChild("HumanoidRootPart")
    if not hrp or not vu547.Character:FindFirstChild("Humanoid") then
        return
    end
    if vu547.Character.Humanoid.Health <= 0 then
        return
    end

    -- agora  seguro acessar p546.Position
    local v548 = (p546.Position - hrp.Position).Magnitude

    local v549 = CheckNearestTeleporter(p546)
    if v549 then
        requestEntrance(v549)
    end

    -- garante que WaitHRP devolve algo vlido antes de usar .CFrame
    local currentHRP = WaitHRP(vu547)
    if not currentHRP then
        -- fallback: usa hrp que j verificamos
        currentHRP = hrp
    end

    if not vu547.Character:FindFirstChild("PartTele") then
        local v550 = Instance.new("Part")
        v550.Size = Vector3.new(10, 1, 10)
        v550.Name = "PartTele"
        v550.Anchored = true
        v550.Transparency = 1
        v550.CanCollide = true
        v550.Parent = vu547.Character
        v550.CFrame = currentHRP.CFrame

        local vu551 = v550

        -- usar a sintaxe de mtodo correta
        vu551:GetPropertyChangedSignal("CFrame"):Connect(function()
            -- upvalues: (ref) vu529, (ref) vu547, (ref) vu551
            if vu529 then
                task.wait()
                if vu547.Character and vu547.Character:FindFirstChild("HumanoidRootPart") then
                    local safeHRP = WaitHRP(vu547) or vu547.Character:FindFirstChild("HumanoidRootPart")
                    if safeHRP then
                        safeHRP.CFrame = vu551.CFrame
                    end
                end
            end
        end)
    end

    vu529 = true

    -- segurana: pega a referncia atual do PartTele
    local partTele = vu547.Character and vu547.Character:FindFirstChild("PartTele")
    if not partTele then
        vu529 = false
        return
    end

    local v553 = TweenService:Create(partTele, TweenInfo.new(v548 / 360, Enum.EasingStyle.Linear), {
        CFrame = p546
    })
    v553:Play()
    v553.Completed:Connect(function(p554)
        -- upvalues: (ref) vu547, (ref) vu529
        if p554 == Enum.PlaybackState.Completed then
            if vu547.Character and vu547.Character:FindFirstChild("PartTele") then
                vu547.Character.PartTele:Destroy()
            end
            vu529 = false
        end
    end)
end

local function fruitHop()
    local response = HttpService:JSONDecode(
        game:HttpGet(
            "https://games.roblox.com/v1/games/" ..
            placeId ..
            "/servers/Public?limit=100&sortOrder=Asc"
        )
    )

    for _, server in pairs(response.data) do
        if server.id ~= game.JobId and server.playing < server.maxPlayers then
            TeleportService:TeleportToPlaceInstance(
                placeId,
                server.id,
                player
            )
            return
        end
    end
end
